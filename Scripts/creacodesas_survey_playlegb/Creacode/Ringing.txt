EVENT CallReject()
{
    string strResultCode = i2str(_RESPONSE_);
    LOG("Creacode-> Ringing:: CallReject in Ringing is interpreting");
    LOG("Creacode-> Ringing:: Reject Cause: " + strResultCode);

    if (g_PlayingRinging == TRUE) {
        g_PlayingRinging = FALSE;
        //StopBackgroundAudio(_LEG_A_);
    }
    
    if (_RESPONSE_ == 404) {
        LOG("Creacode-> Ringing:: Called number is not found");
    }
    if (_RESPONSE_ == 486) {
        LOG("Creacode-> Ringing:: Called number is busy");
    }
    if ((_RESPONSE_ == 408) || (_RESPONSE_ == 480) || (_RESPONSE_ == 484) || (_RESPONSE_ == 487) || (_RESPONSE_ == 500)) {
        LOG("Creacode-> Ringing:: Called number does not answer");
    }
    if (_RESPONSE_ == 603) {
        LOG("Creacode-> Ringing:: called party declined");
    }
    
    UpdateTelephoneStatus(strResultCode);
    
    AbortSession();
}

EVENT CallEnd()
{
    LOG("Creacode-> Ringing:: CallEnd in Ringing is interpreting");
    if (g_PlayingRinging == TRUE) {
        StopBackgroundAudio(_LEG_A_);
    }

    CloseSession();
}
